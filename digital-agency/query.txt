-- Fiverr-style negotiation flow schema updates
-- Safe version for MariaDB / MySQL where some columns may already exist.

ALTER TABLE orders
    ADD COLUMN IF NOT EXISTS client_budget DECIMAL(10,2) NULL DEFAULT NULL AFTER detailed_desc,
    ADD COLUMN IF NOT EXISTS offered_price DECIMAL(10,2) NULL DEFAULT NULL AFTER client_budget,
    ADD COLUMN IF NOT EXISTS payment_mode ENUM('full','milestone') NULL DEFAULT NULL AFTER offered_price,
    ADD COLUMN IF NOT EXISTS offer_status ENUM('waiting','offered','countered','accepted','rejected') NOT NULL DEFAULT 'waiting' AFTER payment_mode,
    ADD COLUMN IF NOT EXISTS milestone_plan LONGTEXT NULL AFTER offer_status;

ALTER TABLE payments
    ADD COLUMN IF NOT EXISTS amount DECIMAL(10,2) NULL DEFAULT NULL AFTER order_id,
    ADD COLUMN IF NOT EXISTS milestone_name VARCHAR(255) NULL DEFAULT NULL AFTER amount,
    ADD COLUMN IF NOT EXISTS note TEXT NULL AFTER milestone_name;

ALTER TABLE payments
    MODIFY COLUMN verified ENUM('pending','yes','no') NOT NULL DEFAULT 'pending';

ALTER TABLE chats
    ADD COLUMN IF NOT EXISTS order_id INT NULL DEFAULT NULL AFTER id;

CREATE TABLE IF NOT EXISTS order_progress (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    progress_percent INT NOT NULL DEFAULT 0,
    message TEXT NOT NULL,
    file VARCHAR(255) DEFAULT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE
);

-- Only run this if `chats.order_id` exists and the foreign key was not created before.
ALTER TABLE chats
    ADD CONSTRAINT fk_chats_order
    FOREIGN KEY (order_id) REFERENCES orders(id) ON DELETE CASCADE;
